46 void Scene::loadScene( std::string toLoad ) {
47  std::ifstream file( toLoad.c_str() );
48  if( !file.is_open() ) {
49  std::cout << "ERROR: Could not open scene file " << toLoad << "\n";
50  }
51  std::string line = "";
52  while( getline( file, line ) ) {
53  switch( line[0] ) {
54  case 'm':
55  loadMesh( line.substr( 2 ) );
56  break;
57  case 't':
58  loadTexture( line.substr( 2 ) );
59  break;
60  case 's':
61  int pos = line.find( ' ', 3 );
62  loadShader( line.substr( 2, pos - 2 ), line.substr( pos ) );
63  break;
64  default:
65  std::cout << "ERROR: Unrecognized data type\n";
66  }
67  }
68  depthMVPs.reserve( lights.size() );
69 }
