457 static unsigned char *convert_format(unsigned char *data, int img_n, int req_comp, uint x, uint y)
458 {
459  int i,j;
460  unsigned char *good;
461 
462  if (req_comp == img_n) return data;
463  assert(req_comp >= 1 && req_comp <= 4);
464 
465  good = (unsigned char *) malloc(req_comp * x * y);
466  if (good == NULL) {
467  free(data);
468  return epuc("outofmem", "Out of memory");
469  }
470 
471  for (j=0; j < (int) y; ++j) {
472  unsigned char *src = data + j * x * img_n ;
473  unsigned char *dest = good + j * x * req_comp;
474 
475  #define COMBO(a,b) ((a)*8+(b))
476  #define CASE(a,b) case COMBO(a,b): for(i=x-1; i >= 0; --i, src += a, dest += b)
477  // convert source image with img_n components to one with req_comp components;
478  // avoid switch per pixel, so use switch per scanline and massive macros
479  switch(COMBO(img_n, req_comp)) {
480  CASE(1,2) dest[0]=src[0], dest[1]=255; break;
481  CASE(1,3) dest[0]=dest[1]=dest[2]=src[0]; break;
482  CASE(1,4) dest[0]=dest[1]=dest[2]=src[0], dest[3]=255; break;
483  CASE(2,1) dest[0]=src[0]; break;
484  CASE(2,3) dest[0]=dest[1]=dest[2]=src[0]; break;
485  CASE(2,4) dest[0]=dest[1]=dest[2]=src[0], dest[3]=src[1]; break;
486  CASE(3,4) dest[0]=src[0],dest[1]=src[1],dest[2]=src[2],dest[3]=255; break;
487  CASE(3,1) dest[0]=compute_y(src[0],src[1],src[2]); break;
488  CASE(3,2) dest[0]=compute_y(src[0],src[1],src[2]), dest[1] = 255; break;
489  CASE(4,1) dest[0]=compute_y(src[0],src[1],src[2]); break;
490  CASE(4,2) dest[0]=compute_y(src[0],src[1],src[2]), dest[1] = src[3]; break;
491  CASE(4,3) dest[0]=src[0],dest[1]=src[1],dest[2]=src[2]; break;
492  default: assert(0);
493  }
494  #undef CASE
495  }
496 
497  free(data);
498  return good;
499 }

549 //
550 // simple implementation
551 // - channel subsampling of at most 2 in each dimension
552 // - doesn't support delayed output of y-dimension
553 // - simple interface (only one output format: 8-bit interleaved RGB)
554 // - doesn't try to recover corrupt jpegs
555 // - doesn't allow partial loading, loading multiple at once
556 // - still fast on x86 (copying globals into locals doesn't help x86)
557 // - allocates lots of intermediate memory (full size of all components)
558 // - non-interleaved case requires this anyway
559 // - allows good upsampling (see next)
560 // high-quality
561 // - upsampled channels are bilinearly interpolated, even across blocks
562 // - quality integer IDCT derived from IJG's 'slow'
563 // performance
564 // - fast huffman; reasonable integer IDCT
565 // - uses a lot of intermediate memory, could cache poorly

567 // stb_jpeg: 1.34 seconds (MSVC6, default release build)

626 static int build_huffman(huffman *h, int *count)
627 {
628  int i,j,k=0,code;
629  // build size list for each symbol (from JPEG spec)
630  for (i=0; i < 16; ++i)
631  for (j=0; j < count[i]; ++j)
632  h->size[k++] = (uint8) (i+1);
633  h->size[k] = 0;
634 
635  // compute actual symbols (from jpeg spec)
636  code = 0;
637  k = 0;
638  for(j=1; j <= 16; ++j) {
639  // compute delta to add to code to compute symbol id
640  h->delta[j] = k - code;
641  if (h->size[k] == j) {
642  while (h->size[k] == j)
643  h->code[k++] = (uint16) (code++);
644  if (code-1 >= (1 << j)) return e("bad code lengths","Corrupt JPEG");
645  }
646  // compute largest code + 1 for this size, preshifted as needed later
647  h->maxcode[j] = code << (16-j);
648  code <<= 1;
649  }
650  h->maxcode[j] = 0xffffffff;
651 
652  // build non-spec acceleration table; 255 is flag for not-accelerated
653  memset(h->fast, 255, 1 << FAST_BITS);
654  for (i=0; i < k; ++i) {
655  int s = h->size[i];
656  if (s <= FAST_BITS) {
657  int c = h->code[i] << (FAST_BITS-s);
658  int m = 1 << (FAST_BITS-s);
659  for (j=0; j < m; ++j) {
660  h->fast[c+j] = (uint8) i;
661  }
662  }
663  }
664  return 1;
665 }

687 // decode a jpeg huffman value from the bitstream
688 __forceinline static int decode(jpeg *j, huffman *h)
689 {
690  unsigned int temp;
691  int c,k;
692 
693  if (j->code_bits < 16) grow_buffer_unsafe(j);
694 
695  // look at the top FAST_BITS and determine what symbol ID it is,
696  // if the code is <= FAST_BITS
697  c = (j->code_buffer >> (j->code_bits - FAST_BITS)) & ((1 << FAST_BITS)-1);
698  k = h->fast[c];
699  if (k < 255) {
700  if (h->size[k] > j->code_bits)
701  return -1;
702  j->code_bits -= h->size[k];
703  return h->values[k];
704  }
705 
706  // naive test is to shift the code_buffer down so k bits are
707  // valid, then test against maxcode. To speed this up, we've
708  // preshifted maxcode left so that it has (16-k) 0s at the
709  // end; in other words, regardless of the number of bits, it
710  // wants to be compared against something shifted to have 16;
711  // that way we don't need to shift inside the loop.
712  if (j->code_bits < 16)
713  temp = (j->code_buffer << (16 - j->code_bits)) & 0xffff;
714  else
715  temp = (j->code_buffer >> (j->code_bits - 16)) & 0xffff;
716  for (k=FAST_BITS+1 ; ; ++k)
717  if (temp < h->maxcode[k])
718  break;
719  if (k == 17) {
720  // error! code not found
721  j->code_bits -= 16;
722  return -1;
723  }
724 
725  if (k > j->code_bits)
726  return -1;
727 
728  // convert the huffman code to the symbol id
729  c = ((j->code_buffer >> (j->code_bits - k)) & bmask[k]) + h->delta[k];
730  assert((((j->code_buffer) >> (j->code_bits - h->size[c])) & bmask[h->size[c]]) == h->code[c]);
731 
732  // convert the id to a symbol
733  j->code_bits -= k;
734  return h->values[c];
735 }

773 static int decode_block(jpeg *j, short data[64], huffman *hdc, huffman *hac, int b)
774 {

792  if (rs < 0) return e("bad huffman code","Corrupt JPEG");
