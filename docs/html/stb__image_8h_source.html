5 //
6 // Limitations:
7 // - no progressive/interlaced support (jpeg, png)
8 // - 8-bit samples only (jpeg, png)
9 // - not threadsafe
10 // - channel subsampling of at most 2 in each dimension (jpeg)
11 // - no delayed line count (jpeg) -- IJG doesn't support either
12 //
13 // Basic usage (see HDR discussion below):
14 // int x,y,n;
15 // unsigned char *data = stbi_load(filename, &x, &y, &n, 0);
16 // // ... process data if not NULL ...
17 // // ... x = width, y = height, n = # 8-bit components per pixel ...
18 // // ... replace '0' with '1'..'4' to force that many components per pixel
19 // stbi_image_free(data)
20 //
21 // Standard parameters:
22 // int *x -- outputs image width in pixels
23 // int *y -- outputs image height in pixels
24 // int *comp -- outputs # of image components in image file
25 // int req_comp -- if non-zero, # of image components requested in result
26 //
27 // The return value from an image loader is an 'unsigned char *' which points
28 // to the pixel data. The pixel data consists of *y scanlines of *x pixels,
29 // with each pixel consisting of N interleaved 8-bit components; the first
30 // pixel pointed to is top-left-most in the image. There is no padding between
31 // image scanlines or between pixels, regardless of format. The number of
32 // components N is 'req_comp' if req_comp is non-zero, or *comp otherwise.
33 // If req_comp is non-zero, *comp has the number of components that _would_
34 // have been output otherwise. E.g. if you set req_comp to 4, you will always
35 // get RGBA output, but you can check *comp to easily see if it's opaque.
36 //
37 // An output image with N components has the following components interleaved
38 // in this order in each pixel:
39 //
40 // N=#comp components
41 // 1 grey
42 // 2 grey, alpha
43 // 3 red, green, blue
44 // 4 red, green, blue, alpha
45 //
46 // If image loading fails for any reason, the return value will be NULL,
47 // and *x, *y, *comp will be unchanged. The function stbi_failure_reason()
48 // can be queried for an extremely brief, end-user unfriendly explanation
49 // of why the load failed. Define STBI_NO_FAILURE_STRINGS to avoid
50 // compiling these strings at all, and STBI_FAILURE_USERMSG to get slightly
51 // more user-friendly ones.
52 //
53 // Paletted PNG and BMP images are automatically depalettized.
54 //
55 //
56 // ===========================================================================
57 //
58 // HDR image support (disable by defining STBI_NO_HDR)
59 //
60 // stb_image now supports loading HDR images in general, and currently
61 // the Radiance .HDR file format, although the support is provided
62 // generically. You can still load any file through the existing interface;
63 // if you attempt to load an HDR file, it will be automatically remapped to
64 // LDR, assuming gamma 2.2 and an arbitrary scale factor defaulting to 1;
65 // both of these constants can be reconfigured through this interface:
66 //
67 // stbi_hdr_to_ldr_gamma(2.2f);
68 // stbi_hdr_to_ldr_scale(1.0f);
69 //
70 // (note, do not use _inverse_ constants; stbi_image will invert them
71 // appropriately).
72 //
73 // Additionally, there is a new, parallel interface for loading files as
74 // (linear) floats to preserve the full dynamic range:
75 //
76 // float *data = stbi_loadf(filename, &x, &y, &n, 0);
77 //
78 // If you load LDR images through this interface, those images will
79 // be promoted to floating point values, run through the inverse of
80 // constants corresponding to the above:
81 //
82 // stbi_ldr_to_hdr_scale(1.0f);
83 // stbi_ldr_to_hdr_gamma(2.2f);
84 //
85 // Finally, given a filename (or an open file or memory block--see header
86 // file for details) containing image data, you can query for the "most
87 // appropriate" interface to use (that is, whether the image is HDR or
88 // not), using:
89 //
90 // stbi_is_hdr(char *filename);
